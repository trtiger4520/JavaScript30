function setClock () {
  let now = new Date;
  let seconds = now.getSeconds();
  let minutes = now.getMinutes();
  let hour = now.getHours();
  let secDeg = ((seconds / 60) * 360) + 90;
  let minDeg = ((minutes / 60) * 360) + 90;
  let hourDeg = ((hour / 12) * 360) + 90;
  //handleSecend.style.transform = `rotate(${secDeg}deg)`;
  setRotate(handleSecend, secDeg);
  setRotate(handleMinutes, minDeg);
  setRotate(handleHour, hourDeg);
}
function setRotate(el, deg) {
  el.style.transform = `rotate(${deg}deg)`;
}
setInterval(setClock, 1000);

function findMatches(word, cities) {
  return cities.filter(place => {
    let regex = new RegExp(word, 'gi');
    return place.city.match(regex) || place.state.match(regex);
  });
}

function updateMatch(e) {
  let value = this.value.trim();
  let filterCityData = findMatches(value, cities);
  let appendHeml = filterCityData.map(citydata => {
    let regex = new RegExp(value, 'gi');
    let cityNameMark = value ? 
      citydata.city.replace(regex, `${ value }`) : 
      citydata.city;
    let stateNameMark = value ? 
      citydata.state.replace(regex, `${ value }`) : 
      citydata.state;
    return `
  • 
        ${ cityNameMark }, ${ stateNameMark }
        ${ citydata.population }
      
    • `;
  }).join('');
  listUL.innerHTML = appendHeml;
}
